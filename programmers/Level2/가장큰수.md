```javascript
function solution(numbers) {
  const zeroCnt = numbers.filter((e) => e == 0).length;

  numbers = numbers.filter((e) => e != 0);

  for (let i = 0; i < Math.max(...numbers).toString().length; i++) {
    numbers.sort((a, b) => {
      let aa = a.toString()[i] * 1;
      let bb = b.toString()[i] * 1;

      if (!a.toString()[i]) aa = a.toString()[i - 1] * 1;
      if (!b.toString()[i]) bb = b.toString()[i - 1] * 1;

      return bb - aa;
    });
  }

  return numbers.join('').concat('0'.repeat(zeroCnt)) == 0 ? '0' : numbers.join('').concat('0'.repeat(zeroCnt));
}
```

function solution(numbers) {
  var answer = numbers
    .map((c) => c + '')
    .sort((a, b) => b + a - (a + b))
    .join('');

  return answer[0] === '0' ? '0' : answer;
}
